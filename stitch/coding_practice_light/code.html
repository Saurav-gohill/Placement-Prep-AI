Valid Parentheses

Medium Stacks

Problem Description

Given a string s containing just the characters '(', ')', '{', '}', '[' and ']', determine if the input string is valid.

Example 1

Input: s = "()"
Output: true

Example 2

Input: s = "()[]{}"
Output: true

Constraints

  • 1 ≤ s.length ≤ 104
  • s consists of parentheses only.

/**
* @param {string} s
* @return {boolean}
*/
var isValid = function(s) {
let stack = [];
const map = {
')': '(',
'}': '{',
']': '['
};
for (let char of s) {
if (map[char]) {
if (stack.pop() !== map[char]) return false;
} else {
stack.push(char);
}
}
return stack.length === 0;
};
